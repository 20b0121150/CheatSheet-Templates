SASS Cheatsheet for Beginners
  1. Sass Variables 
                        
                        $red: #833;
                        body {
                          color: $red;
                        }
  2. Commenets 
                      
                    /* multiline
                     comments */
                    // Line comments
  3. Nesting 
                    
                    .main {
                        a {
                          color: blue;
                          &:hover {
                            color: red;
                          }
                        }
                      }

                      // In properties

                      text: {
                        align: center;          // like text-align: center
                        transform: uppercase;   // like text-transform: uppercase
                      }
  4. Mixins 
                    
                    @mixin main-font {
                        font-family: sans-serif;
                        font-weight: bold;
                      }
                      h1 {
                        @include main-font;
                      }
  5. Extend 
                    
                    .button {
                        ···
                      }
                      .cta-button {
                        @extend .button;
                      }

                      // with parameters

                        @mixin font-size($dfs) {
                        font-size: $dfs * 1.5em;
                        }
                        body {
                        @include font-size(2);
                        }

                      // with default values
                        @mixin pad($pd: 1rem) {
                        padding: $pd;
                        }
                        body {
                        @include pad(1rem);
                        }

                       // with a default variable

                        // Set a default value
                        $default-padding: 2rem;
                        @mixin pad($pd: $default-padding) {
                        padding: $pd;
                        }
                        body {
                        @include pad(2rem);
                        }
  6. Composing 
                    
                    @import './other_sass_file';
  7. The len() Function 
                    
                >>> len('hello')
                5
  8. Mixing 
                    
                    mix($a, $b, 10%)   // 10% a, 90% b
  9. Getting Individual Values 
                    
                     // HSLA

                    hue($color)         // → 0deg..360deg
                    saturation($color)  // → 0%..100%
                    lightness($color)   // → 0%..100%
                    alpha($color)       // → 0..1 (opacity())

                     // RGB

                    red($color)         // → 0..255
                    green($color)
                    blue($color)
  10. Modifying HSLA 
                    
                    darken($color, 5%)
                    lighten($color, 5%)

                    saturate($color, 5%)
                    desaturate($color, 5%)
                    grayscale($color)

                    adjust-hue($color, 15deg)
                    complement($color)    // like adjust-hue(_, 180deg)
                    invert($color)

                    fade-in($color, .5)   // aka opacify()
                    fade-out($color, .5)  // aka transparentize() - halves the opacity
                    rgba($color, .5)      // sets alpha to .5
  11. Adjustments 
                    
                    // Changes by fixed amounts

                    adjust-color($color, $blue: 5)
                    adjust-color($color, $lightness: -30%)   // like darken(_, 30%)
                    adjust-color($color, $alpha: -0.4)       // like fade-out(_, .4)
                    adjust-color($color, $hue: 30deg)        // like adjust-hue(_, 15deg)

                    // Changes via percentage

                    scale-color($color, $lightness: 50%)

                    // Changes one property completely

                    change-color($color, $hue: 180deg)
                    change-color($color, $blue: 250)
  12. Strings 
                    
               unquote('hello')
               quote(hello)

               to-upper-case(hello)
               to-lower-case(hello)

               str-length(hello world)
               str-slice(hello, 2, 5)      // "ello" - it's 1-based, not 0-based
               str-insert("abcd", "X", 1)  // "Xabcd"

  16. Misc

                    selector-nest('.menu', '&:hover li')  // .menu:hover li
                    selector-extend(...)
                    selector-parse(...)
                    selector-replace(...)
                    selector-unify(...)

  18. For Loops 
                    
                    @for $i from 1 through 4 {
                        .item-#{$i} { left: 20px * $i; }
                      }
  19. Each loops(simple) 
                    
                    $menu-links: home about works contact;

                    @each $link in $menu-links {
                      .photo-#{$link} {
                        background: url('images/#{$link}.jpg');
                      }
                    }
  20. Each Loops (Nested) 
                    
                    $backgrounds: (home, 'home.jpg'), (about, 'about.jpg');

                    @each $id, $image in $backgrounds {
                      .photo-#{$id} {
                        background: url($image);
                      }
                    }
  21. While loops 
                    
                    $num: 8;
                    @while $num > 0 {
                      .item-#{$num} { width: 2em * $num; }
                      $num: $num - 2;
                    }
  22. Conditionals 
                    
                    @if $position == 'left' {
                        position: absolute;
                        left: 0;
                     }
                     @else if $position == 'right' {
                        position: absolute;
                        right: 0;
                     }
                     @else {
                        position: static;
                     }
  23. Interpolation 
                    
                    .#{$klass} { ... }      // Class
                    call($function-name)    // Functions
                    
                    @media #{$tablet}
                    font: #{$size}/#{$line-height}
                    url("#{$background}.jpg")
  24. Maps

                    map-get($map, key1)
  25. Lists
