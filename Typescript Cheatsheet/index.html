This Typescript Cheatsheet For You!
  1. Basic Types 
    
                        let varName: any;       // Untyped, it could be any type
                        let varName: object;    // A non primitve value
                        let varName: undefined; // An undefined value 
                        let varName: null;      // A null value
                        let varName: unknown;   // A value with an unknown type
                        let varName: string;    // A string
                        let varName: number;    // A number that can be float or int
                        let varName: boolean;   // A true or false value
  2. Example to define an Array 
    
                        string[] or Array<string>  // An array of strings
                        number[] or Array<number>  // An array of numbers
                        any[]    or Array<any>     // An array of anything
  3. Example of Object types 
    
                        {
                            id: number;

                        }
  4. Tuples

  5. Enums 
    
                        enum Options {

                            COMPUTED = getValue()
                        }
  6. Union and intersection types 
    
                        let unionVar: number | boolean;
                        let intersectionVar: Type1 & Type2;
  7. Functions 
    
                        (arg1: Type): ReturnType => { ...; return value; }      // Arrow function
                        (arg: Type, argN: Type) => Type;                        // Function type
                        (arg: Type, optional?: Type) => ReturnType              // Function type with optional param 
                        (arg: Type, ...options: Type[]) => ReturnType           // Function type with rest param 
                        function nameFn(arg = 'default'): ReturnType {}         // Default argument
  8. Class 
    
                        class HelloWorld {
                            constructor() {}
                            hello(arg: string) {
                                console.log(arg);
                            }
                        }

                        // Example
                        const sayHello = new HelloWorld();
                        sayHello("Hello World");     // Pints in the console "hello world"
  9. Access Modifires 
    
                        class HelloWorld {
                            property: Type;                                        // By default the property is public 
                            public defaultProperty = 'default value';              // A property public means that it can be readed outside of the class
                            private _privateProperty: Type;                        // A property that can only be readed in the class
                            private readonly _privateReadonlyProperty: Type;       // A property that can only be readed in the class and immutable
                            static staticProperty: Type;                           // A property that can be used without making a new instance of the class
                            protected protectedProperty: Type                      // Members are only visible to subclasses of the class they're declared in
                        }
  10. Getters / Setters 
    
                        class HelloWorld {
                            private _length = 0;
                            get length() {
                              return this._length;    // Gets the value from _length
                            }
                            set length(value) {
                              this._length = value;   // Sets the value from _length
                            }
                        }
  11. Class heritage with implements 
    
                        interface Animal {
                            makeNoise(): void;
                          }
                          // You can use an implements clause to check that a class satisfies a particular interface.
                          class Dog implements Animal {
                            makeNoise() {
                              console.log("waf waf");
                            }
                          }
                          // Classes may also implement multiple interfaces, e.g. class C implements A, B
  12. Class heritage with extends 
    
                        class Animal {
                            move() {
                              console.log("Moving along!");
                            }
                          }
                           
                          class Dog extends Animal {
                            makeNoise() {
                                console.log("woof!");
                            }
                          }

                          const myNewDog = new Dog();
                          // Base class method
                          myNewDog.move();
                          // Derived class method
                          myNewDog.makeNoise();
  13. Generic classes 
    
                        class Box<Type>  {
                            contents: Type;
                            constructor(value: Type) {
                              this.contents = value;
                            }
                          }
                           
                          const newBox = new Box("hello!");
                          // newBox will be type Box<string>

  15. Interface

                        // Example

  16. Some Utility Types 
    
                      // Partial
                      Partial<{ x: number;  }>                                 // same as { x?: number; }
                      
                      // Readonly 
                      Readonly<{ x: number; }>                                //  same as { readonly x: }
  
                      // Pick
                      Pick<{ x: number; y: number; z: number; }, 'x' | 'y'>  // same as { x: number; y: number; }

                      // Record 
                      Record<'x' | 'y' | 'z', number>                        // same as { x: number; y: number; z: number; }
