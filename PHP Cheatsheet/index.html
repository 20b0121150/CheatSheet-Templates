This PHP Cheatsheet is For You!
  1. Print output on screen 
    <?php echo "Hello World!" ?>
  2. Integer Variable 
    <?php $a = 3 ?>
  3. Float Variable 
    <?php $a = 4.6 ?>
  4. String Variable 
    <?php $a = "Hello World"  ?>
  5. Array 
    <?php $cars = array("BMW", "Jaguar", "Ferrai")  ?>
  6. Associative Array 
    <?php $person = array(
              'name' => 'John',
              'age' => 34
            ) ?>
  7. Class 
    <?php  class Cars {
            //code goes here...  
          }  ?>
  8. If statement 
     <?php if(condition) {
            //code goes here...  
          } ?>
  9. If..Else 
    <?php if(condition) {
            // if condition is met  
          } else {
            // if condition is not met
          } ?>
  10. If..Elseif..Else 
    if(condition) {
              // if this condition is met
            } else if(condition) {
              //if this condition is met
            } else {
              // if none of the condition is met
            }
  11. Switch Statement 
    <?php switch(num) {
            case x:
              // code goes here... 
            break;
            case y:
              //code goes here...
            break;
            default:
              code goes here...
          } ?>
  12. For Loop 
    <?php for(initialization; condition; increment/decrement) {
            //code goes here...   
          } ?>
  13. While Loop 
    <?php while(condition) {
            // code goes here...
            increment/decrement  
          } ?>
  14. Do-While 
    <?php do {
            //code goes here...  
          }while(condition); ?>
  15. Foreach Loop 
    <?php foreach(arrayName: value) {
            //code goes here...  
          } ?>
  16. Filename of currently executing script. 
     <?php $_SERVER['PHP_SELF'] ?>
  17. IP Address of host server 
    <?php $_SERVER['SERVER_ADDR'] ?>
  18. Name of the host server 
    <?php $_SERVER['SERVER_NAME'] ?>
  19. Request Method 
     $_SERVER['REQUEST_METHOD']
  20. Get Method 
    <?php $_GET['variable_name'] ?>
  21. Post Method 
    <?php $_POST['variable_name'] ?>
  22. isset 
    <?php if(isset($_POST['variable_name'])) {
            //code goes here...   
          } ?>
